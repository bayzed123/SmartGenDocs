     <pre id="code-python" class="text-sm text-gray-300 whitespace-pre leading-relaxed">import os, requests, google.generativeai as genai

REPO = open("config.txt").read().strip()
OWNER, NAME = REPO.split("/")[-2], REPO.split("/")[-1]
HEADERS = {"Authorization": f"token {os.environ['GH_TOKEN']}"}

def fetch(path):
    r = requests.get(f"https://api.github.com/repos/{OWNER}/{NAME}/contents/{path}", headers=HEADERS)
    return r.json().get("content","") if r.ok else ""

readme = fetch("README.md")
pkg = fetch("package.json") or fetch("requirements.txt")
workflow = fetch(".github/workflows")

genai.configure(api_key=os.environ["GEMINI_KEY"])
model = genai.GenerativeModel("gemini-1.5-flash")

prompts = {
    "overview.md": f"Write a project overview based on:\\n{readme}",
    "architecture.md": f"Describe the architecture from:\\n{readme}\\n{pkg}",
    "setup.md": f"Rewrite setup instructions uniquely from:\\n{readme}\\n{pkg}"
}

for fname, prompt in prompts.items():
    resp = model.generate_content(prompt)
    with open(f"docs/{fname}", "w") as f:
        f.write(resp.text)

links = {"overview.md":"architecture.md","architecture.md":"setup.md","setup.md":"overview.md"}
for src, dst in links.items():
    with open(f"docs/{src}","a") as f:
        f.write(f"\\n\\n---\\n[Next → {dst}](/{dst})")

os.system('git add docs/ &amp;&amp; git commit -m "Auto-generate docs" &amp;&amp; git push')</pre>
     <pre id="code-yaml" class="text-sm text-gray-300 whitespace-pre leading-relaxed hidden">name: SmartGen Docs Pipeline

on:
  push:
    branches: [main]
    paths: ['config.txt']

jobs:
  generate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install deps
        run: pip install requests google-generativeai

      - name: Generate documentation
        env:
          GH_TOKEN: ${{ secrets.GH_TOKEN }}
          GEMINI_KEY: ${{ secrets.GEMINI_KEY }}
        run: |
          mkdir -p docs
          python scripts/generate_docs.py

      - name: Commit &amp; push
        run: |
          git config user.name "SmartGen Bot"
          git config user.email "bot@smartgen.dev"
          git add docs/
          git commit -m "docs: auto-generated" || exit 0
          git push</pre>
